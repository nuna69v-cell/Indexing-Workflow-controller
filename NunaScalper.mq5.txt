#property link      "https://www.mql5.com"
#property version   "1.10"
#property description "Scalping Strategy for M5, M15, M30 timeframes based on EMA and MACD."

#include <Trade\Trade.mqh>

//--- Input parameters
input int      InpMagicNum    = 123456;      // Magic Number
input double   InpLots        = 0.01;        // Lot Size (if Risk=0)
input double   InpRiskPercent = 1.0;         // Risk Percent per trade (0 to disable)
input int      InpEMAPeriod   = 20;          // EMA Period
input int      InpMACDFast    = 12;          // MACD Fast EMA
input int      InpMACDSlow    = 26;          // MACD Slow EMA
input int      InpMACDSignal  = 9;           // MACD Signal
input int      InpATRPeriod   = 14;          // ATR Period
input double   InpSLMultiplier= 1.5;         // Stop Loss ATR Multiplier
input double   InpTPMultiplier= 2.0;         // Take Profit ATR Multiplier
input double   InpMaxSpread   = 3.0;         // Max Spread (pips)
input int      InpStartHour   = 8;           // Start Trading Hour
input int      InpEndHour     = 20;          // End Trading Hour
input bool     InpUseTrailing = true;        // Use Trailing Stop
input int      InpTrailStart  = 10;          // Trailing Start (pips)
input int      InpTrailStep   = 5;           // Trailing Step (pips)

//--- Global variables
CTrade         trade;
int            handle_ema;
int            handle_macd;
int            handle_atr;
datetime       last_candle_time;
double         PipSize;

//+------------------------------------------------------------------+
//| Expert initialization function                                   |
//+------------------------------------------------------------------+
int OnInit()
  {
   //--- Initialize trade object
   trade.SetExpertMagicNumber(InpMagicNum);

   //--- Initialize Pip Size
   if(_Digits == 3 || _Digits == 5)
      PipSize = 10.0 * _Point;
   else
      PipSize = _Point;

   //--- Initialize indicators
   handle_ema = iMA(_Symbol, _Period, InpEMAPeriod, 0, MODE_EMA, PRICE_CLOSE);
   handle_macd = iMACD(_Symbol, _Period, InpMACDFast, InpMACDSlow, InpMACDSignal, PRICE_CLOSE);
   handle_atr = iATR(_Symbol, _Period, InpATRPeriod);

   if(handle_ema == INVALID_HANDLE || handle_macd == INVALID_HANDLE || handle_atr == INVALID_HANDLE)
     {
      Print("Error creating indicator handles");
      return(INIT_FAILED);
     }

   last_candle_time = 0;

   Print("NunaScalper Initialized on ", _Symbol, " ", EnumToString(_Period));
   return(INIT_SUCCEEDED);
  }

//+------------------------------------------------------------------+
//| Expert deinitialization function                                 |
//+------------------------------------------------------------------+
void OnDeinit(const int reason)
  {
   IndicatorRelease(handle_ema);
   IndicatorRelease(handle_macd);
   IndicatorRelease(handle_atr);
   Print("NunaScalper Deinitialized");
  }

//+------------------------------------------------------------------+
//| Expert tick function                                             |
//+------------------------------------------------------------------+
void OnTick()
  {
   //--- Check for open positions and manage trailing stop
   ManageTrailingStop();

   //--- Check for new bar to avoid tick spamming
   datetime time[];
   ArraySetAsSeries(time, true);
   if(CopyTime(_Symbol, _Period, 0, 1, time) < 1) return;

   if(time[0] == last_candle_time)
      return;

   last_candle_time = time[0];

   //--- Filter: Trading Hours
   MqlDateTime dt;
   TimeCurrent(dt);
   if(dt.hour < InpStartHour || dt.hour >= InpEndHour)
      return;

   //--- Filter: Spread
   long current_spread_points = SymbolInfoInteger(_Symbol, SYMBOL_SPREAD);
   double max_spread_points = InpMaxSpread * (PipSize / _Point);

   if(current_spread_points > max_spread_points)
      return;

   //--- Get Indicator Data
   double ema[], macd_main[], macd_signal[], atr[];
   ArraySetAsSeries(ema, true);
   ArraySetAsSeries(macd_main, true);
   ArraySetAsSeries(macd_signal, true);
   ArraySetAsSeries(atr, true);

   if(CopyBuffer(handle_ema, 0, 0, 3, ema) < 3 ||
      CopyBuffer(handle_macd, 0, 0, 3, macd_main) < 3 ||
      CopyBuffer(handle_macd, 1, 0, 3, macd_signal) < 3 ||
      CopyBuffer(handle_atr, 0, 0, 1, atr) < 1)
     {
      Print("Error copying indicator buffers");
      return;
     }

   //--- Strategy Logic (Close of previous candle [1])
   double close[];
   ArraySetAsSeries(close, true);
   if(CopyClose(_Symbol, _Period, 0, 3, close) < 3) return;

   bool buy_signal = (close[1] > ema[1]) && (macd_main[1] > macd_signal[1]) && (macd_main[1] > 0);
   bool sell_signal = (close[1] < ema[1]) && (macd_main[1] < macd_signal[1]) && (macd_main[1] < 0);

   // Check if we already have a position
   if(PositionsTotal() > 0)
     {
      for(int i = PositionsTotal() - 1; i >= 0; i--)
        {
         ulong ticket = PositionGetTicket(i);
         if(PositionSelectByTicket(ticket))
           {
            if(PositionGetString(POSITION_SYMBOL) == _Symbol && PositionGetInteger(POSITION_MAGIC) == InpMagicNum)
               return; // Already in a trade, don't open another
           }
        }
     }

   //--- Calculate Lot Size
   double lot = InpLots;
   if(InpRiskPercent > 0)
     {
      double balance = AccountInfoDouble(ACCOUNT_BALANCE);
      double risk_amount = balance * (InpRiskPercent / 100.0);
      double sl_dist_price = atr[0] * InpSLMultiplier;

      double tick_size = SymbolInfoDouble(_Symbol, SYMBOL_TRADE_TICK_SIZE);
      double tick_value = SymbolInfoDouble(_Symbol, SYMBOL_TRADE_TICK_VALUE);

      if(sl_dist_price > 0 && tick_size > 0 && tick_value > 0)
        {
         double money_loss_per_lot = (sl_dist_price / tick_size) * tick_value;
         lot = risk_amount / money_loss_per_lot;
        }

      // Check min/max/step lots
      double min_lot = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MIN);
      double max_lot = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MAX);
      double vol_step = SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP);

      if(vol_step > 0)
         lot = MathFloor(lot / vol_step) * vol_step;

      if(lot < min_lot) lot = min_lot;
      if(lot > max_lot) lot = max_lot;

      // Safety check if risk calculation resulted in absurdly low lot (below min) which is handled above,
      // but if lot was > max, we cap it.
     }

   //--- Open Orders
   double sl, tp;
   double current_ask = SymbolInfoDouble(_Symbol, SYMBOL_ASK);
   double current_bid = SymbolInfoDouble(_Symbol, SYMBOL_BID);

   if(buy_signal)
     {
      sl = current_ask - (atr[0] * InpSLMultiplier);
      tp = current_ask + (atr[0] * InpTPMultiplier);
      // Normalize SL/TP
      sl = NormalizeDouble(sl, _Digits);
      tp = NormalizeDouble(tp, _Digits);

      if(!trade.Buy(lot, _Symbol, current_ask, sl, tp, "NunaScalper Buy"))
        {
         Print("Buy Order Failed. Return Code: ", trade.ResultRetcode(), " Desc: ", trade.ResultRetcodeDescription());
        }
     }
   else if(sell_signal)
     {
      sl = current_bid + (atr[0] * InpSLMultiplier);
      tp = current_bid - (atr[0] * InpTPMultiplier);
      // Normalize SL/TP
      sl = NormalizeDouble(sl, _Digits);
      tp = NormalizeDouble(tp, _Digits);

      if(!trade.Sell(lot, _Symbol, current_bid, sl, tp, "NunaScalper Sell"))
        {
         Print("Sell Order Failed. Return Code: ", trade.ResultRetcode(), " Desc: ", trade.ResultRetcodeDescription());
        }
     }
  }

//+------------------------------------------------------------------+
//| Manage Trailing Stop                                             |
//+------------------------------------------------------------------+
void ManageTrailingStop()
  {
   if(!InpUseTrailing) return;

   for(int i = PositionsTotal() - 1; i >= 0; i--)
     {
      ulong ticket = PositionGetTicket(i);
      if(PositionSelectByTicket(ticket))
        {
         if(PositionGetString(POSITION_SYMBOL) == _Symbol && PositionGetInteger(POSITION_MAGIC) == InpMagicNum)
           {
            double current_sl = PositionGetDouble(POSITION_SL);
            double current_price = PositionGetDouble(POSITION_PRICE_CURRENT);
            double open_price = PositionGetDouble(POSITION_PRICE_OPEN);
            double tp = PositionGetDouble(POSITION_TP);

            if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_BUY)
              {
               if(current_price - open_price > InpTrailStart * PipSize)
                 {
                  double new_sl = current_price - (InpTrailStep * PipSize);
                  new_sl = NormalizeDouble(new_sl, _Digits);

                  if(new_sl > current_sl && new_sl < current_price)
                    {
                     if(!trade.PositionModify(ticket, new_sl, tp))
                        Print("Modify Position Failed. Ret: ", trade.ResultRetcode(), " Desc: ", trade.ResultRetcodeDescription());
                    }
                 }
              }
            else if(PositionGetInteger(POSITION_TYPE) == POSITION_TYPE_SELL)
              {
               if(open_price - current_price > InpTrailStart * PipSize)
                 {
                  double new_sl = current_price + (InpTrailStep * PipSize);
                  new_sl = NormalizeDouble(new_sl, _Digits);

                  if((new_sl < current_sl || current_sl == 0) && new_sl > current_price)
                    {
                     if(!trade.PositionModify(ticket, new_sl, tp))
                        Print("Modify Position Failed. Ret: ", trade.ResultRetcode(), " Desc: ", trade.ResultRetcodeDescription());
                    }
                 }
              }
           }
        }
     }
  }
